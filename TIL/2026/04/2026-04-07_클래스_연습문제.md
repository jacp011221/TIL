```java
// BankApplication
package ch06.sec00.test20;

import java.util.Scanner;

public class BankApplication {
    private Account[] accounts = new Account[100];
    private int count;

    public void print() {
        System.out.println("----------");
        System.out.println("계좌목록");
        System.out.println("----------");
        for(int i=0; i<count; i++) {
            System.out.print(accounts[i].getNum() + "\t" +
                    accounts[i].getName() + "\t" +
                    accounts[i].getMoney() + "\n");
        }
    }

    public void set(Account accounts) {
        this.accounts[count] = accounts;
        // 만약 계좌가 이미 100개가 생성된 상태라면 예외가 발생한다.
        // 따라서 다음과 같이 작성하는 것이 안전하다.
        // for(int i=0; i<this.accounts.length; i++) {
        //     if(this.accounts[i] == null) {
        //         this.accounts[i] = accounts;
        //         break;
        //     }
        // }
        count += 1;
    }

    public void plus(String num) {
        int cheak = -1;
        Scanner scanner = new Scanner(System.in);

        for(int i=0; i<count; i++) {
            if(accounts[i].getNum().equals(num)) {
                cheak = i;
            }
        }
        if (cheak == -1) {
            System.out.println("일치하는 계좌번호가 없습니다.");
            return;
        }
        System.out.print("예금액: ");
        accounts[cheak].setMoney(Integer.parseInt(scanner.nextLine()));
    }

    public void minus(String num) {
        int cheak = -1;
        Scanner scanner = new Scanner(System.in);

        for(int i=0; i<count; i++) {
            if(accounts[i].getNum().equals(num)) {
                cheak = i;
            }
        }
        if (cheak == -1) {
            System.out.println("일치하는 계좌번호가 없습니다.");
            return;
        }
        System.out.print("출금액: ");
        accounts[cheak].setMoney(
                ~( Integer.parseInt(scanner.nextLine()) - 1 )
        );
    }
}
```

```java
package ch06.sec00.test20;

import java.util.Scanner;

public class test {
    public static void main(String[] args) {
        BankApplication bankApplication = new BankApplication();

        system: while(true) {
            System.out.println("---------------------------------------------");
            System.out.println("1.계좌생성 | 2.계좌목록 | 3.예금 | 4.출금 | 5.종료");
            System.out.println("---------------------------------------------");
            System.out.print("선택> ");
            Scanner scanner = new Scanner(System.in);
            String str1 = scanner.nextLine();

            switch (str1) {
                case "1":
                    Account account = new Account();
                    bankApplication.set(account);
                    System.out.println("결과: 계좌가 생성되었습니다.");
                    break;
                case "2":
                    bankApplication.print();
                    break;
                case "3":
                    System.out.println("----------");
                    System.out.println("예금");
                    System.out.println("----------");
                    System.out.print("계좌번호: ");
                    bankApplication.plus(scanner.nextLine());
                    break;
                case "4":
                    System.out.println("----------");
                    System.out.println("출금");
                    System.out.println("----------");
                    System.out.print("계좌번호: ");
                    bankApplication.minus(scanner.nextLine());
                    System.out.println("결과: 출금이 성공되었습니다.");
                    break;
                case "5":
                    System.out.println("프로그램 종료");
                    break system;
            }
        }
    }
}
```
