```java
package ch05.sec00;

public class Ex5_8 {
    public static void main(String[] args) {
        int[][] array = {
                {95, 86},
                {83, 92, 96},
                {78, 83, 93, 87, 88}
        };

        int sum = 0;
        int num = 0;

        for(int i=0; i<array.length; i++) {
            for(int k=0; k<array[i].length; k++) {
                sum += array[i][k];
            }
            num += array[i].length;
        }

        System.out.println(sum);
        System.out.println( (double) sum / num);
    }
}
```

# 5-9.

```java
package ch05.sec00;

import java.util.Scanner;

public class Ex5_9 {
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);

        int[] scores = null;
        int sum = 0;

        boolean cheak1 = false;
        boolean cheak2 = false;
        boolean cheak3 = false;

        system: while(true) {

            System.out.println("---------------------------------------------");
            System.out.println("1.학생수 | 2.점수입력 | 3.점수리스트 | 4.분석 | 5.종료");
            System.out.println("---------------------------------------------");
            System.out.print("선택> ");
            String str1 = scan.nextLine();
            String str2;

            switch(str1) {
                case "1":
                    System.out.print("학생수> ");
                    str2 = scan.nextLine();
                    scores = new int[Integer.parseInt(str2)];
                    cheak1 = true;
                    cheak2 = false;
                    cheak3 = false;
                    sum = 0;
                    break;
                case "2":
                    if(cheak1 != true) break;
                    for(int i=0; i<scores.length; i++) {
                        System.out.print("scores[" + i + "]> ");
                        str2 = scan.nextLine();
                        scores[i] = Integer.parseInt(str2);
                        sum += scores[i];
                    }
                    cheak2 = true;
                    cheak3 = false;
                    break;
                case "3":
                    if(cheak2 != true) break;
                    for(int i=0; i<scores.length; i++) {
                        System.out.println("scores[" + i + "]: " + scores[i]);
                    }
                    cheak3 = true;
                    break;
                case "4":
                    if(cheak3 != true) break;
                    int best = 0;
                    for(int i=0; i<scores.length; i++) {
                        if( best < scores[i] ) {
                            best = scores[i];
                        }
                    }
                    System.out.println("최고 점수: " + best);
                    System.out.println("평균 점수: " + (double) sum / scores.length);
                    break;
                case "5":
                    System.out.println("프로그램 종료");
                    break system;
            }
        }
    }
}
```
