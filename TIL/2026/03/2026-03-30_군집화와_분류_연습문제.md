# 12-5.

```r
library(class)
library(mlbench)
data("Soybean")

mydata <- Soybean[complete.cases(Soybean),]
idx <- c()
acc <- c()

for(i in 1:562){
  if(i%%2==0){
    idx[i/2] <- i
  }
}

ds.tr <- mydata[idx,2:36]
ds.ts <- mydata[-idx,2:36]
cl.tr <- factor(mydata[idx,1])
cl.ts <- factor(mydata[-idx,1])

for(i in 1:3){
  pred <- knn(ds.tr,ds.ts,cl.tr,k=i*2+1,prob=T)
  acc[i] <- mean(pred==cl.ts)
}

acc
```

# 12-6.

```r
library(class)
library(cvTools)
library(mlbench)
data("Glass")

mydata <- Glass
k <- 5
folds <- cvFolds(nrow(mydata),K=k)
acc <- c()

for(i in 1:k){
  ts.idx <- folds$which==i
  ds.tr <- mydata[-ts.idx,1:9]
  ds.ts <- mydata[ts.idx,1:9]
  cl.tr <- mydata[-ts.idx,10]
  cl.ts <- mydata[ts.idx,10]
  
  pred <- knn(ds.tr,ds.ts,cl.tr,k=3,prob=T)
  acc[i] <- mean(pred==cl.ts)
}

mean(acc)
```
