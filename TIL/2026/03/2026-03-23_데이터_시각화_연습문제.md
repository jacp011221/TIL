# 8-1.

```r
# 1번
library(treemap)

us$name <- rownames(us)
treemap(us,
        index=c("state.division","name"),
        vSize="Population",
        vColor="Income",
        type="value")
# East North Central이 가장 인구가 많으며, Alaska는 인구는 적지만 소득은 가장 높다.

# 2번
treemap(us,
        index=c("state.division","name"),
        vSize="HS.Grad",
        vColor="Murder",
        type="value")
# Mountain이 가장 고등학교 졸업률이 높으며, East South Central은 전반적으로 범죄율이 높다.

# 3번
symbols(us$Incom,us$Illiteracy,
        circles=us$Population,
        bg="green")
text(us$Income,us$Illiteracy,
     us$name)
# 소득이 높을수록 문맹률이 낮은 경향을 보이며,
# 대략 반 이상의 인구가 소득이 높은 지역에 거주하고 있는 것으로 추정된다.

# 4번
symbols(us$Illiteracy,us$Murder,
        circles=us$Area,
        bg="green")
text(us$Illiteracy,us$Murder,
     us$name)
# 문맹률이 높을수록 범죄율이 상승하는 경향을 보인다.
```

```r
sw$group <- "MID"

treemap(sw,
        index=c("group","name"),
        vSize="Fertility",
        vColor="Agriculture",
        type="value")
```

```r

library(Rtsne)
library(car)
library(rgl)
library(mgcv)

x77 <- data.frame(state.x77)

tsne <- Rtsne(x77,dims=2,perplexity=10)
x77_tsne <- data.frame(tsne$Y)

ggplot(x77_tsne,aes(X1,X2)) +
  geom_point()
```
