# 4-9.

```r
lgm <- function(x,y){
  xlist <- c()
  xi <- 1
  ylist <- c()
  yn <- 1
  for(i in x:1){
    if(x%%i==0){
      xlist[xi] <- i
      xi <- xi + 1
    }
  }
  for(n in y:1){
    if(y%%n==0){
      ylist[yn] <- n
      yn <- yn + 1
    }
  }
  for(xl in 1:length(xlist)){
    for(yl in 1:length(ylist)){
      if(xlist[xl]==ylist[yl]){
        return(xlist[xl])
      }
    }
  }
}
```
