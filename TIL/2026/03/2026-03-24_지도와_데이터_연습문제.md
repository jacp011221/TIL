# 9-2.

```r
library(ggmap)
library(ggplot2)
register_google(key="구글키")

name <- c("신사역","논현역","신논현역","강남역","양재역","양재시민의숲역","청계산입구역",
          "판교역","정자역","미금역","동천역","수지구청역","성복역","상현역","광교중앙역","광교역")
gc <- geocode(enc2utf8(name))
df <- data.frame(name,gc)
cen <- c(mean(gc$lon),mean(gc$lat))

map <- get_googlemap(center=cen,
                     maptype="roadmap",
                     zoom=11,
                     marker=gc)
ggmap(map) +
  geom_text(data=df,aes(x=lon,y=lat),
            size=4,
            label=df$name)
```

```r
library(ggmap)
library(ggplot2)
register_google(key="구글키")

setwd("작업 폴더의 경로 이름")
da <- read.csv("도로교통공단_시도_시군구별_월별_교통사고(2018).csv",header=T,fileEncoding = "euc-kr")
da2 <- aggregate(da[,4:9],
                 by=list(시도=da$시도,월=da$월),
                 FUN=sum)
da3 <- da2[-which(da2$시도=="세종")]
da4 <- da3[which(da3$월==c("07월","08월")),]
da5 <- aggregate(da4[,3:6],
                 by=list(시도=da4$시도),
                 FUN=sum)

gc <- geocode(enc2utf8(da5$시도))
df <- data.frame(da5,gc)
cen <- as.numeric(colMeans(gc))

map <- get_googlemap(center=cen,
                     type="roadmap",
                     zoom=7)

  geom_point(data=df,
             aes(x=lon,y=lat,size=부상자수),
             alpha=0.5,
             col="blue") +
```
