pid <- PimaIndiansDiabetes

# 2번
model <- glm(diabetes~.,data=train)

coef(model)[1] + coef(model)[2]*pregnant + coef(model)[3]*glucose + coef(model)[4]*pressure +
  coef(model)[5]*triceps + coef(model)[6]*insulin + coef(model)[7]*mass + coef(model)[8]*pedigree +
  coef(model)[9]*age

pred <- predict(model,test)
pred <- round(pred,0)

# 4번
abs(mean(pred-test$diabetes))
