text3 <- readLines("ex_10_3.txt",encoding = "UTF-8")
buildDictionary(ext_dic="woorimalsam")
pal2 <- brewer.pal(8,"Dark2")

noun1 <- sapply(text1,extractNoun,USE.NAMES=F)
noun2 <- sapply(text2,extractNoun,USE.NAMES=F)
noun3 <- sapply(text3,extractNoun,USE.NAMES=F)
noun1_2 <- unlist(noun1)
noun2_2 <- unlist(noun2)
noun3_2 <- unlist(noun3)

noun1_2 <- noun1_2[nchar(noun1_2)>1]
noun2_2 <- noun2_2[nchar(noun2_2)>1]
noun3_2 <- noun3_2[nchar(noun3_2)>1]

wordcount1 <- table(noun1_2)
wordcount2 <- table(noun2_2)
wordcount3 <- table(noun3_2)

set.seed(101)
wordcloud(names(wordcount1),
          freq=wordcount1,
          scale=c(6,0.7),
          min.freq=3,
          random.order=F,
          rot.per=0.1,
          colors=pal2)
set.seed(101)

          freq=wordcount2,
          scale=c(6,0.7),

ds <- read.csv("Online_Retail.csv")

tans <- read.transactions("Online_Retail_upd.csv",format="single",
                          header=T,cols=c(1,2),
                          sep=",",rm.duplicates=T)
tans

# 2번
dimnames(tans)[2]

# 3번
inspect(head(tans,10))

# 4번
itemFrequencyPlot(tans,topN=10,type="absolute",
                  col="green")

# 5번
rules <- apriori(tans,parameter=list(supp=0.004,conf=0.7))

# 6번
rules.sort <- sort(rules,by="confidence",decreasing=T)
inspect(head(rules.sort,10))

cen <- as.numeric(colMeans(gc))

map <- get_googlemap(center=cen,
                     maptype="roadmap",
